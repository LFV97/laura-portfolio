import { useEffect } from "react";
import { create } from "zustand";
import { Button } from "@/components/ui/button";
import i18n from "i18next";
import { initReactI18next, useTranslation } from "react-i18next";

// Initialize i18n
i18n.use(initReactI18next).init({
  resources: {
    en: {
      translation: {
        switchTheme: "Switch to",
        rpgMode: "RPG Mode",
        classicMode: "Classic Mode",
        welcomeClassic: "Welcome to my portfolio in a clean, professional layout.",
        welcomeRPG: "This is the RPG-themed version of my portfolio.",
        language: "Español",
      },
    },
    es: {
      translation: {
        switchTheme: "Cambiar a",
        rpgMode: "Modo RPG",
        classicMode: "Modo Clásico",
        welcomeClassic: "Bienvenido a mi portafolio en un diseño limpio y profesional.",
        welcomeRPG: "Esta es la versión con temática RPG de mi portafolio.",
        language: "English",
      },
    },
  },
  lng: localStorage.getItem("portfolioLanguage") || "es",
  fallbackLng: "es",
  interpolation: { escapeValue: false },
});

// Zustand store for global state management
const useStore = create((set) => ({
  theme: localStorage.getItem("portfolioTheme") || "classic",
  toggleTheme: () =>
    set((state) => {
      const newTheme = state.theme === "classic" ? "rpg" : "classic";
      localStorage.setItem("portfolioTheme", newTheme);
      return { theme: newTheme };
    }),
  toggleLanguage: () => {
    const newLanguage = i18n.language === "es" ? "en" : "es";
    i18n.changeLanguage(newLanguage);
    localStorage.setItem("portfolioLanguage", newLanguage);
  },
}));

export default function Portfolio() {
  const { theme, toggleTheme, toggleLanguage } = useStore();
  const { t } = useTranslation();

  return (
    <div className={`min-h-screen flex items-center justify-center ${theme === "rpg" ? "bg-gray-900 text-white" : "bg-white text-black"}`}>
      <div className="absolute top-4 right-4 flex gap-4">
        <Button onClick={toggleTheme} className="px-4 py-2 rounded-lg">
          {t("switchTheme")} {theme === "classic" ? t("rpgMode") : t("classicMode")}
        </Button>
        <Button onClick={toggleLanguage} className="px-4 py-2 rounded-lg">
          {t("language")}
        </Button>
      </div>

      {theme === "rpg" ? (
        <div className="p-8 text-center">
          <h1 className="text-4xl font-bold">{t("rpgMode")}</h1>
          <p className="mt-4">{t("welcomeRPG")}</p>
        </div>
      ) : (
        <div className="p-8 text-center">
          <h1 className="text-4xl font-bold">{t("classicMode")}</h1>
          <p className="mt-4">{t("welcomeClassic")}</p>
        </div>
      )}
    </div>
  );
}
